<script>

// 1. tạo map (tọa độ Hà Nội)
var map = L.map('map').setView([21.03, 105.85], 11);

// 2. thêm nền OpenStreetMap
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '© OpenStreetMap'
}).addTo(map);

// 3. đọc file GeoJSON
fetch('data.geojson')
  .then(function(response) {
    return response.json();
  })
  .then(function(data) {

    // 4. vẽ lên bản đồ
    L.geoJSON(data, {
      style: {
        color: "blue",
        weight: 2,
        fillOpacity: 0.5
      },

      onEachFeature: function(feature, layer) {
        // hiển thị tên quận nếu có
        if (feature.properties) {
          layer.bindPopup(feature.properties.name || "District");
        }
      }

    }).addTo(map);

  });

</script>
